
Question 2

In a given array of unsorted numbers, remove the duplicates in the array without using extra array

Program
#include <stdio.h>
int main()
{
    //Badhri : Avoid x,y,z,a,b,c,n variable names
    int n,i,j,num, count = 1;
    scanf("%d",&n);
    int a[n];
    for(i = 0; i < n; i++) {
        scanf("%d",&num);
        if(i > 0) {
            for(j = 0; j < i; j++) {
                if(num == a[j])
                    break;
            }
            if(j == i) {
                a[count] = num;
                count++;
            }
        }
        else
            a[i] = num;
    }
    for(i = 0; i < count; i++)
        printf("%d ",a[i]);
    return 0;
}
