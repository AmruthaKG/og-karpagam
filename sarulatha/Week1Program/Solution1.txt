Question 1

Given a set of intervals, merge all the overlapping set of intervals.
For example, 
Input
[5,15], [23,35], [55,80], [30,50]
Output 
[5,15],[23,50],[55,80]


Program
/* considering [2,3] [3,4] as an overlap */
#include <stdio.h>
int main()
{
    int n,i,j;
    scanf("%d",&n);
    int a[n][2];
    for(i = 0; i < n; i++)
        scanf("%d,%d",&a[i][0],&a[i][1]);
    for(i = 0; i < n; i++) {
       for(j = 0; j < n; j++) {
           if(a[j][0] != 0 && a[j][1] != 0) {
           if(a[j][0] > a[i][0] && a[j][0] <= a[i][1]) {
               if(a[j][1] > a[i][1])
                a[i][1] = a[j][1];
               a[j][0] = a[j][1] = 0;
               j = 0;
           }
        }
       }
    }
    for(i = 0; i < n; i++) {
        if(a[i][0] != 0 && a[i][1] != 0)
            printf("[%d,%d]\n",a[i][0],a[i][1]);
    }
    return 0;
}
